    inl player_ac {net name rate} cd =
        open Learning
        inl agent = 
			rl
				.with_context cd
				.initialize net {Observation Action}
				.with_rate rate
			
        inl methods = {basic_methods with
            bet=inl s input -> s.agent.action input
            showdown=inl s r -> 
				s.agent.reward r
				s.agent.backward
				s.agent.optimize
				s.agent.reset // Could be truncate.
            game_over=inl s -> ()
            }

        Object
            .member_add methods
            .data_add {name; win=ref 0; agent}
